    (x, y) = (y, x) if x > y else (x, y)
    for factor in range(x, 0, -1):

def is_palindrome(num):
    temp = num
    total = 0
    while temp > 0:
        total = total * 10 + temp % 10
        temp //= 10
    return num == total
